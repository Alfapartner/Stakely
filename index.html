    <script>

        try {

            const urlParams = new URLSearchParams(window.location.search);

            const tcLink = urlParams.get('tc'); // Parameter name is 'tc'


            if (tcLink) {

                const decodedTcLink = decodeURIComponent(tcLink);

                console.log("Redirecting to:", decodedTcLink);

                window.location.href = decodedTcLink; // Redirect

            } else {

                document.body.innerHTML = '<p style="color: red;">Error: Connection link parameter is missing.</p>';

                console.error("Missing 'tc' parameter in URL");

            }

        } catch (e) {

             document.body.innerHTML = '<p style="color: red;">Error initiating wallet connection.</p>';

             console.error("Error during redirect:", e);

        }

    </script>
