    <script>

        try {

            // Get the full query string (e.g., "?tc=tc%3A%2F%2F...")

            const queryString = window.location.search;

            // Create URLSearchParams object

            const urlParams = new URLSearchParams(queryString);

            // Get the value of the 'tc' parameter

            const tcLink = urlParams.get('tc'); // Assuming we name the parameter 'tc'


            if (tcLink) {

                // Decode the tcLink just in case it was double-encoded, though usually not needed if encoded once correctly

                const decodedTcLink = decodeURIComponent(tcLink);

                console.log("Redirecting to:", decodedTcLink);

                // Redirect the browser to the tc: link

                window.location.href = decodedTcLink;

                // Optionally, try to close the window after a delay if it's a popup scenario, though often not needed/possible

                // setTimeout(() => window.close(), 2000);

            } else {

                document.body.innerHTML = '<p style="color: red;">Error: Connection link parameter is missing.</p>';

                console.error("Missing 'tc' parameter in URL");

            }

        } catch (e) {

             document.body.innerHTML = '<p style="color: red;">Error initiating wallet connection.</p>';

             console.error("Error during redirect:", e);

        }

    </script>
